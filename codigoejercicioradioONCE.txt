    if (radioEnMetros > 0) { //Se utiliza un if para verificar si el radio ingresado es mayor que cero. Esto garantiza que el radio cumpla con la restricción de ser positivo.

    // Paso 3: Calcular el área del círculo usando la fórmula A = pi * radio^2. donde pi es:
    const PI = 3.14;

    const areaEnMetrosCuadrados = PI * Math.pow(radioEnMetros, 2); //Se utiliza una constante PI para almacenar el valor de pi (aproximadamente 3.14). Luego, se calcula el área en metros cuadrados usando la fórmula especificada.

    // Paso 4: Convertir el área a centímetros cuadrados (1 metro cuadrado = 10,000 cm cuadrados). //Se multiplica el área en metros cuadrados por 10,000 para convertirla a centímetros cuadrados, ya que 1 metro cuadrado equivale a 10,000 centímetros cuadrados.
    const areaEnCentimetrosCuadrados = areaEnMetrosCuadrados * 10000; //Se multiplica el área en metros cuadrados por 10,000 para convertirlo a centímetros cuadrados, ya que 1 metro cuadrado equivale a 10.000 centímetros cuadrados.

    // Paso 5: Mostrar el resultado en centímetros cuadrados en consola.
    console.log(`El área del círculo es: ${areaEnCentimetrosCuadrados} cm²`); //Se utiliza console.log para mostrar el resultado en la consola del navegador

} else {

    console.log("Error: El radio debe ser mayor que cero.");
